
district_gensokyo_jinja = {
	base_buildtime = 600
	is_capped_by_modifier = no
	icon = district_gensokyo_jinja
	
	show_on_uncolonized = {
		OR = {
			uses_district_set = gensokyo_city
		}
	}
	
	potential = {
		uses_district_set = gensokyo_city
	}
	
	conversion_ratio = 1
	convert_to = {
		district_city
		district_nexus
		district_hive
	}
	
	resources = {
		category = planet_districts
		cost = {
			minerals = 15000
			sr_lingli = 5500
		}
		upkeep = {
			sr_lingli = 35
		}		
	}
	
	planet_modifier = {
		planet_housing_add = 3
		job_reimu_add = 1
		job_sanae_add = 1
		job_Immortal_add = 2
	}

	ai_weight = {
		weight = 0
		modifier = {
			weight = 500
			owner = {
				has_monthly_income = {
					resource = alloys
					value < 500
				}
				has_monthly_income = {
					resource = minerals
					value > 99
				}
			}
		}
	}
}
