district_gensokyo_renli = {
	base_buildtime = 600
	is_capped_by_modifier = no
	icon = district_gensokyo_renli
	
	show_on_uncolonized = {
		uses_district_set = gensokyo_planet
	}

	zone_slots = {
		slot_gensokyo_renli
	}
	potential = {
		OR = {
			uses_district_set = gensokyo_planet
			uses_district_set = gensokyo_city
		}
	}
	
	conversion_ratio = 1
	convert_to = {
		district_city
		district_nexus
		district_hive
	}

	resources = {
		category = planet_districts
		cost = {
			minerals = 2000
			sr_lingli = 500
		}
		produces = {
			food = 10
			trigger = {
				OR = {
					colony_type = col_spth_basis
					colony_type = col_spth_basis_city
				}
			}
		}
		produces = {
			exotic_gases = 0.6
			volatile_motes = 0.6
			rare_crystals = 0.6
			trigger = {
				OR = {
					colony_type = col_spth_refine
					colony_type = col_spth_refine_city
				}
			}
		}
		upkeep = {
			energy = 15
			sr_lingli = 5
		}
		upkeep = {
			food = 10
			trigger = {
				OR = {
					colony_type = col_spth_refine
					colony_type = col_spth_refine_city
				}
			}
		}
	}

	planet_modifier = {
		planet_housing_add = 4000
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_fallen_empire = no
				is_regular_empire = yes
				has_authority = auth_gensokyo
			}
		}
		modifier = {
			job_spth_story_farmer_add = 400
			job_spth_story_chemist_add = 200
		}
	}
	triggered_planet_modifier = {
		potential = { check_modifier_value = { modifier = planet_spth_has_gensokyo_city_foods_zone value > 1 } }
		job_spth_story_farmer_add = 200
		job_spth_story_chemist_add = -200
		job_spth_story_chemist_add = -200
	}
	triggered_planet_modifier = {
		potential = { check_modifier_value = { modifier = planet_spth_has_gensokyo_city_refine_zone value > 1 } }
		job_spth_story_farmer_add = -400
		job_spth_story_chemist_add = 100
	}

	ai_weight = {
		weight = 2
		modifier = {
			weight = 5
			free_housing < 10
		}
	}
}
