ship_behavior = {
	name = "spth_line"
	preferred_attack_range = 180		# max/min/median/12/412..这是船舶在攻击时想要保持的目标距离
	formation_distance = 180			# 当轨道/跟随船将尝试与指定位置保持此距离
	return_to_formation_distance = 200	# 攻击时，如果到指定位置的距离大于此值，船舶将切换到被动模式

	collision_awareness = 1
	collision_radius = 1
	collision_strafe = 3				#collision_strafe allows the ship to move in any direction while colliding with another

	attack_move_pattern = charge
	passive_move_pattern = stay_at_range

	desc = "SPTH_LINE_BEHAVIOR_DESC"
}

ship_behavior = {
	name = "spth_flagship"
	preferred_attack_range = 400			#舰船进攻模式下保持的进攻距离 #PS：舰船处于进攻模式下会保持该距离
	formation_distance = 400				#舰船被动模式下保持的攻击距离，如果小于该距离切换为进攻模式 #PS:当舰船处于被动模式将会保持该距离，一旦运动到低于该距离，强制转化为进攻模式
	return_to_formation_distance = 450		#抵达该距离之前，强制转化为被动模式 #PS:即任何比该数值要大的距离和包括该距离都是被动模式，包括进战
	source_target_anchor_factor = 1.0		#目标和锚点之间坐标权重倍率，该值不能为0 不写该条目默认为1，最低值疑似为0.1或者0.01 数值越低锚点离自己舰船越近
	combat_target_anchor = root				#以敌人舰队领袖船为目标锚点 parent是锁定目标敌舰 root是敌人舰队首位船

	collision_awareness = 0.1 				#舰船运动意识，当舰船的移动半径运算结果要撞上去之前，有多高的移动概率，越大的船移动概率越低
	collision_radius = 0.5					#舰船移动半径倍率，与舰船的定义半径进行相乘，这一数值应该优先修正舰船shipsize定义中的半径，使其与舰船的大小相近
	collision_strafe = 1.2					#舰船位置冲突时，舰船会选择多大的角度进行回避，一般情况下拉高点舰船回避的速度越快，容易出现鬼畜，数值要适合
	attack_move_pattern = stationary			#舰船进攻模式
	passive_move_pattern = charge		#舰船被动模式

	desc = "SPTH_CARRIER_BEHAVIOR_DESC"
}
