		modifier = {
			factor = 0.1
			has_citizenship_rights = no
			NOT = { has_trait = trait_mechanical }
		}
		modifier = {
			factor = 5
			has_job = miko
		}
		modifier = {
			factor = 1.2
			has_trait = trait_touhou
		}
		# crisis purge
		modifier = {
			factor = 0.01
			exists = planet

			planet.controller = {
				OR = {
					is_country_type = swarm

				}
			}
		}
	}
