namespace = th_info_display

event = {
	id = th_info_display.0
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		any_country = {
			has_authority = auth_gensokyo
		}
	}

		every_country = {
			limit = {
				has_authority = auth_gensokyo
			}

		}
	}
}
