country_event = {
	id = th_weapon_up.1
	hide_window = yes

	is_triggered_only = yes
	trigger = {
		has_authority = auth_gensokyo
		FROMFROM = {
			exists = leader
			leader = {
				leader_class = admiral
				has_leader_flag = th_sp_leader
			}
		}
		NOT = { has_modifier = spth_p_max }
	}
	immediate = {
		change_variable = {
			which = spth_number_p
			value = 2
		}
		if = {
			limit = {
				check_variable = {
					which = spth_number_p
					value = 8
				}
			}

			country_event = {
				id = th_weapon_up.2
			}
		}
		
		else_if = {
			limit = {
				check_variable = {
					which = spth_number_p
					value = 16
				}
			}

			}
		}
		else_if = {

				}
			}

			}

			country_event = {

		}
		else_if = {

			country_event = {

country_event = {
	id = th_weapon_up.10
	title = "th_weapon_up.10.name"
	desc = {
		text = th_weapon_up.10.first.desc
		trigger = { NOT = { has_country_flag = spth_weapon_had_up } }
	}
	desc = {
		text = th_weapon_up.10.had.desc
		trigger = { 
				has_country_flag = spth_weapon_had_up
		}
	}
	
	
	diplomatic_title = BLANK_STRING
	diplomatic = yes
	force_open = yes
	
	picture_event_data = {

		room = spth_muq_room
	}
	
	is_triggered_only = yes

	trigger = {
		has_authority = auth_gensokyo
	}
	###初次介绍
	option = { 
		custom_gui = "enclave_curator_option"
		trigger = { 
			NOT = {
				has_country_flag = spth_weapon_had_up
			}
		}
		name = th_weapon_up.10.a
		hidden_effect = {
			set_country_flag = spth_weapon_had_up
		}
		is_dialog_only = yes
		response_text = th_weapon_up.10.a.response
	}

	###升级
	option = {
		custom_gui = "enclave_curator_option"
		trigger = {
			has_modifier = spth_p_max
		}
		name = th_weapon_up.10.b
		hidden_effect = {
			if = {
				limit = {NOT = {has_country_flag = spth_weapon_had_up}}
				set_country_flag = spth_weapon_had_up
			}
			country_event = { id = th_weapon_up.11 }
		}
	}
		
	###退出
	option = { 
		custom_gui = "enclave_curator_option"
		name = th_weapon_up.10.z
		hidden_effect = {
			if = {
				limit = {NOT = {has_country_flag = spth_weapon_had_up}}
				set_country_flag = spth_weapon_had_up
			}
			country_event = { id = th_weapon_up.20 }
		}
	}
}
