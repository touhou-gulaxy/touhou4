			}
		}
		owner = { add_resource = { energy = -1000 volatile_motes = -500 } }

			limit = { has_astral_planes_dlc = yes }

	}
	option = {
		name = "th_arc.210.b"
		if = {
			limit = { has_astral_planes_dlc = yes }

		} else = { small_artifact_reward = yes }

	}
}
