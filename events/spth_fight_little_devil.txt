	id = spth_fight_little_devil.11
	hide_window = yes

		exists = leader

			leader_class = commander

			name = "NAME_spth_yeguai"

					who = owner
					location = owner.capital_scope
				}

					class = commander
					species = owner_species

						trait = random_trait
						trait = random_trait
					}
				}

					name = "NAME_spth_yeguai_1"
					effect = {
						set_owner = prev
						create_ship = {
							name = "NAME_Unyielding_Faith"
							design = "NAME_th_yeguai_1"
						}
						create_ship = {
							name = "NAME_Holy_Avatar"
							design = "NAME_th_yeguai_1"
						}
						create_ship = {
							name = "NAME_Divine_Fury"
							design = "NAME_th_yeguai_1"
						}
						set_location = {
							target = event_target:th_yeguai_fleet_planet
							distance = 80
							angle = random
						}
						set_fleet_stance = aggressive
						set_aggro_range_measure_from = return_point
						set_aggro_range = 200
						assign_leader = last_created_leader
					}
				}
				while = {
					count = 9
					create_leader = {
						class = commander
						species = owner_species
						name = random
						skill = 2
						traits = {
							trait = random_trait
							trait = random_trait
						}
					}
					create_fleet = {
						name = "NAME_spth_yeguai_1"
						effect = {
							set_owner = prev
							create_ship = {
								name = "NAME_Unyielding_Faith"
								design = "NAME_th_yeguai_1"
							}
							create_ship = {
								name = "NAME_Holy_Avatar"
								design = "NAME_th_yeguai_1"
							}
							create_ship = {
								name = "NAME_Divine_Fury"
								design = "NAME_th_yeguai_1"
							}
							set_location = {
								target = event_target:th_yeguai_fleet_planet
								distance = 80
								angle = random
							}
							set_fleet_stance = aggressive
							set_aggro_range_measure_from = return_point
							set_aggro_range = 200
							assign_leader = last_created_leader
						}
					}
				}
				save_event_target_as = th_yeguai
				set_name = random
				set_country_flag = th_yeguai_of@root
			}
		}
		owner = {
			country_event = {
