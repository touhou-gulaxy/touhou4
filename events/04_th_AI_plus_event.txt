	id = th_AI_plus.1
	hide_window = yes
	is_triggered_only = yes

		owner = {
			is_ai = yes
			has_authority = auth_gensokyo
		}
	}
	immediate = {
		if = {
			limit = { has_building = building_colony_shelter }
			remove_building = building_colony_shelter
			add_building = building_capital
		}
		if = {
			limit = { NOT = {has_building = building_mls_1 }}
			add_building = building_mls_1
			while = {
				count = 2
				create_pop = {
					species = owner_species
				}
			}
		}
	}
}

country_event = {
	id = th_AI_plus.4
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		NOT = {end_game_years_passed >= 0}
		NOT = {
			has_global_flag = spth_AI_NOT_plus
		}
		owner = {
			is_ai = yes
			has_authority = auth_gensokyo
		}
		years_passed >= 10
		count_owned_fleet = {
			count <= 40
		}
	}
	immediate = {
		random_owned_starbase = {
			save_event_target_as = spth_ship_loc
		}
		if = {
			limit = { 
				OR = {
					years_passed = 30
					years_passed = 40
					years_passed = 50
					years_passed = 60
					years_passed = 70
					years_passed = 80
					years_passed = 90
				}
			}
			while = {
				count = 5
				random_owned_fleet = {
					limit = {
						is_ship_class = shipclass_military
					}
					destroy_fleet = { kill_leader = yes destroy_template = yes target = this }
				}
				
			}
		}
		if = {
			limit = {
				resource_stockpile_compare = {
					resource = alloys
					value >= 500
				}
				has_monthly_income = { resource = energy value >= 100 }
				used_naval_capacity_percent <= 1
			}
			create_fleet = {
				set_take_point = yes
				effect = {
					set_owner = prev
					while = {
						count = 10
						create_ship = {
							name = random
							random_existing_design = corvette
						}
					}
					set_location = event_target:spth_ship_loc
				}
			}
